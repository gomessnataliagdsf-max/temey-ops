EMPRESAS ───────────────────────────────────────────── CREATE TABLE IF NOT EXISTS empresas ( id UUID DEFAULT gen_random_uuid() PRIMARY KEY, nome TEXT NOT NULL, codigo TEXT NOT NULL UNIQUE, -- ex: GRUPOD-MATRIZ cnpj TEXT, ativa BOOLEAN DEFAULT true, grupo_id UUID REFERENCES empresas(id) ON DELETE SET NULL, -- para agrupar filiais criado_em TIMESTAMPTZ DEFAULT NOW() ); -- ── 2. USUÁRIOS ───────────────────────────────────────────── -- (sua tabela sistema_usuarios já existe — apenas adicionamos perfil e ativo) -- Se precisar recriar do zero: CREATE TABLE IF NOT EXISTS sistema_usuarios ( id UUID DEFAULT gen_random_uuid() PRIMARY KEY, nome TEXT NOT NULL, login TEXT NOT NULL UNIQUE, senha_hash TEXT NOT NULL, perfil TEXT NOT NULL DEFAULT 'assistente', -- perfis: master | admin | coordenador | assistente | auxiliar ativo BOOLEAN DEFAULT true, empresa_id UUID REFERENCES empresas(id), -- empresa "base" do usuário (opcional) criado_em TIMESTAMPTZ DEFAULT NOW() ); -- ── 3. VÍNCULO USUÁRIO ↔ EMPRESAS (N:N) ──────────────────── CREATE TABLE IF NOT EXISTS usuarios_empresas ( id UUID DEFAULT gen_random_uuid() PRIMARY KEY, usuario_id UUID NOT NULL REFERENCES sistema_usuarios(id) ON DELETE CASCADE, empresa_id UUID NOT NULL REFERENCES empresas(id) ON DELETE CASCADE, criado_em TIMESTAMPTZ DEFAULT NOW(), UNIQUE(usuario_id, empresa_id) -- evita duplicatas ); -- Índices para performance CREATE INDEX IF NOT EXISTS idx_ue_usuario ON usuarios_empresas(usuario_id); CREATE INDEX IF NOT EXISTS idx_ue_empresa ON usuarios_empresas(empresa_id); -- ── 4. VAGAS (já existente — garantir empresa_id) ─────────── -- ALTER TABLE vagas ADD COLUMN IF NOT EXISTS empresa_id UUID REFERENCES empresas(id); -- ── 6.
